int run(List<List<int>> board) {
  if (board.length != sz || board[0].length != sz) {
    return 0;
  }

  dlx.init(9 * 9 * 9, 9 * 9 * 4);

  for (int i = 1; i <= 9; i++) {
    for (int j = 1; j <= 9; j++) {
      RangeError.checkValueInInterval(board[i][j], 0, 10);

      if (board[i][j] != 0) {
        insert(i, j, board[i][j]);
      } else {
        for (int v = 1; v <= 9; v++) {
          insert(i, j, v);
        }
      }
    }
  }

  // 0, 1, 2
  return dlx.dance(1, board);
}
