static List<List<int>> toBoard(String s) {
  if (s.length != 81) {
    throw ArgumentError.value(s.length, 'string length error != 81');
  }
  List<List<int>> b =
      List.generate(10, (index) => List.generate(10, (index) => 0));
  for (int i = 0; i < 9; i++) {
    for (int j = 0; j < 9; j++) {
      int id = i * 9 + j;
      b[i + 1][j + 1] = int.parse(s[id]);
    }
  }
  return b;
}
